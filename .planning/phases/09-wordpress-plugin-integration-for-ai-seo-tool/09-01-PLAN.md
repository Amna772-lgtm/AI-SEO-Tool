def create_api_key_record(key_id: str, user_id: str, key_hash: str, name: str, created_at: str) -> None:
    """Insert a new API key record. Raw key is NOT stored — only the SHA-256 hash."""
    with _lock:
        conn = _connect()
        try:
            conn.execute(
                "INSERT INTO api_keys (id, user_id, key_hash, name, created_at) VALUES (?, ?, ?, ?, ?)",
                (key_id, user_id, key_hash, name, created_at),
            )
            conn.commit()
        finally:
            conn.close()


def list_api_keys(user_id: str) -> list[dict[str, Any]]:
    """Return all API keys for a user. Never returns key_hash."""
    conn = _connect()
    try:
        rows = conn.execute(
            "SELECT id, name, created_at, last_used_at FROM api_keys WHERE user_id = ? ORDER BY created_at DESC",
            (user_id,),
        ).fetchall()
        return [dict(r) for r in rows]
    finally:
        conn.close()


def revoke_api_key(key_id: str, user_id: str) -> bool:
    """Delete an API key by id, scoped to user. Returns True if deleted."""
    with _lock:
        conn = _connect()
        try:
            cursor = conn.execute(
                "DELETE FROM api_keys WHERE id = ? AND user_id = ?",
                (key_id, user_id),
            )
            conn.commit()
            return cursor.rowcount > 0
        finally:
            conn.close()


def get_user_by_api_key_hash(key_hash: str) -> dict[str, Any] | None:
    """Look up user via API key hash. Returns full user dict or None."""
    conn = _connect()
    try:
        row = conn.execute(
            "SELECT u.* FROM users u JOIN api_keys ak ON u.id = ak.user_id WHERE ak.key_hash = ?",
            (key_hash,),
        ).fetchone()
        return dict(row) if row else None
    finally:
        conn.close()


def update_api_key_last_used(key_hash: str) -> None:
    """Stamp last_used_at on an API key."""
    with _lock:
        conn = _connect()
        try:
            now = datetime.now(timezone.utc).isoformat()
            conn.execute(
                "UPDATE api_keys SET last_used_at = ? WHERE key_hash = ?",
                (now, key_hash),
            )
            conn.commit()
        finally:
            conn.close()
```

@router.post("/api-key", status_code=status.HTTP_201_CREATED)
def create_api_key(
    body: ApiKeyCreateRequest,
    current_user: dict[str, Any] = Depends(get_current_user),
) -> dict[str, Any]:
    """Generate a new API key. Returns raw key ONCE — only hash stored in DB."""
    raw_key = secrets.token_urlsafe(32)
    key_hash = hashlib.sha256(raw_key.encode()).hexdigest()
    key_id = str(uuid.uuid4())
    created_at = datetime.now(timezone.utc).isoformat()
    create_api_key_record(key_id, current_user["id"], key_hash, body.name, created_at)
    return {
        "id": key_id,
        "key": raw_key,
        "name": body.name,
        "created_at": created_at,
    }


@router.get("/api-keys", response_model=list[ApiKeyOut])
def get_api_keys(
    current_user: dict[str, Any] = Depends(get_current_user),
) -> list[dict[str, Any]]:
    """List all API keys for the current user. Never returns raw key or hash."""
    return list_api_keys(current_user["id"])


@router.delete("/api-keys/{key_id}", status_code=status.HTTP_200_OK)
def delete_api_key(
    key_id: str,
    current_user: dict[str, Any] = Depends(get_current_user),
) -> dict[str, str]:
    """Revoke an API key by ID."""
    deleted = revoke_api_key(key_id, current_user["id"])
    if not deleted:
        raise HTTPException(
            status_code=status.HTTP_404_NOT_FOUND,
            detail="API key not found",
        )
    return {"status": "revoked"}
```

"""Tests for API key generation, listing, revocation, and Bearer auth."""
import hashlib
from fastapi.testclient import TestClient
from app.main import app

client = TestClient(app)


def _create_user_and_signin(email: str = "apitest@test.com", name: str = "API Tester", password: str = "testpass123") -> TestClient:
    """Helper: signup + signin, return client with auth cookie set."""
    client.post("/auth/signup", json={"email": email, "name": name, "password": password})
    resp = client.post("/auth/signin", json={"email": email, "password": password})
    assert resp.status_code == 200
    return client


def test_create_api_key():
    """POST /auth/api-key returns id, key, name, created_at. (WP-01)"""
    c = _create_user_and_signin("create_key@test.com")
    resp = c.post("/auth/api-key", json={"name": "Test Key"})
    assert resp.status_code == 201
    data = resp.json()
    assert "id" in data
    assert "key" in data
    assert data["name"] == "Test Key"
    assert "created_at" in data
    assert len(data["key"]) >= 40  # token_urlsafe(32) produces 43 chars


def test_list_api_keys():
    """GET /auth/api-keys returns keys without raw key or hash. (WP-02)"""
    c = _create_user_and_signin("list_keys@test.com")
    c.post("/auth/api-key", json={"name": "Key A"})
    c.post("/auth/api-key", json={"name": "Key B"})
    resp = c.get("/auth/api-keys")
    assert resp.status_code == 200
    keys = resp.json()
    assert len(keys) >= 2
    for k in keys:
        assert "key" not in k
        assert "key_hash" not in k
        assert "name" in k
        assert "id" in k


def test_revoke_api_key():
    """DELETE /auth/api-keys/{id} revokes the key. (WP-03)"""
    c = _create_user_and_signin("revoke_key@test.com")
    create_resp = c.post("/auth/api-key", json={"name": "Revoke Me"})
    key_id = create_resp.json()["id"]
    resp = c.delete(f"/auth/api-keys/{key_id}")
    assert resp.status_code == 200
    assert resp.json()["status"] == "revoked"
    # Verify it's gone from the list
    list_resp = c.get("/auth/api-keys")
    ids = [k["id"] for k in list_resp.json()]
    assert key_id not in ids


def test_bearer_auth_on_me():
    """Bearer API key accepted on /auth/me endpoint. (WP-04)"""
    c = _create_user_and_signin("bearer_test@test.com")
    raw_key = c.post("/auth/api-key", json={"name": "Bearer Test"}).json()["key"]
    # Clear cookies to prove Bearer works independently
    c.cookies.clear()
    resp = c.get("/auth/me", headers={"Authorization": f"Bearer {raw_key}"})
    assert resp.status_code == 200
    assert resp.json()["email"] == "bearer_test@test.com"


def test_cookie_auth_still_works():
    """Cookie JWT auth still works after dual-auth change. (WP-05)"""
    c = _create_user_and_signin("cookie_test@test.com")
    # No Bearer header — just cookie
    resp = c.get("/auth/me")
    assert resp.status_code == 200
    assert resp.json()["email"] == "cookie_test@test.com"


def test_invalid_key_returns_401():
    """Invalid Bearer token returns 401. (WP-06)"""
    resp = client.get("/auth/me", headers={"Authorization": "Bearer invalid-key-abc123"})
    assert resp.status_code == 401
